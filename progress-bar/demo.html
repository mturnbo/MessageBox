Progress Bar

Polls an API at a given interval. Fills its container. Stops at 100%.

// 1. Load assets
<link rel="stylesheet" href="progress-bar.css">
<script src="progress-bar.js"></script>

// 2. Give your container an explicit height (any width works)
<div id="my-bar" style="height:24px;border-radius:6px;position:relative"></div>

// 3. Create
const bar = createProgressBar(document.getElementById('my-bar'), {
  endpoint: '/api/progress', // returns { progress: 0–100 }
  interval: 1000, // ms between polls
  color: '#4f46e5', // any CSS colour
});

// 4. Control
bar.stop(); // pause polling
bar.resume(); // restart polling
bar.setProgress(42); // set manually (stops polling at 100)
bar.destroy(); // remove from DOM + stop polling
